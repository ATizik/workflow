WorkflowActionTester

public struct WorkflowActionTester<WorkflowType, Action> where WorkflowType == Action.WorkflowType, Action : WorkflowAction

Testing helper that chains event sending and state/output assertions to make tests easier to write.

reducer
    .tester(withState: .firstState)
    .assertState { state in
        XCTAssertEqual(.firstState, state)
    }
    .send(event: .exampleEvent) { output in
        XCTAssertEqual(.finished, output)
    }
    .assertState { state in
        XCTAssertEqual(.differentState, state)
    }
